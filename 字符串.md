# 字符串变形 BM83  
将字符大小写反转，并且按照空格，把字符分开以后倒着排。
```cpp  
  class Solution {
  public:
      string trans(string s, int n) {
          vector<string> vs;
          string result;
          int fast = 0;
          int slow = 0;
          while (fast < n){
              while (s[fast] != ' ' && fast < n){
                  if (isupper(s[fast])) s[fast] = tolower(s[fast]);
                  else  s[fast] = toupper(s[fast]);
                  fast++;
              }
              vs.push_back(s.substr(slow, fast - slow));
              if (s[fast] == ' ') vs.push_back(" ");
              fast++;
              slow = fast;
          }
          reverse(vs.begin(), vs.end());
          for (auto v : vs) {
              result += v;
          }

          return result;
      }
  };
```
