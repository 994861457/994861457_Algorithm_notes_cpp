# 1.分糖果 BM95  
```cpp
  int candy(vector<int>& arr) {
            vector<int> results(arr.size());
            results[0] = 1;

            for (int i = 1; i < arr.size(); i++){
                if (arr[i] > arr[i - 1]){
                    results[i] = results[i - 1] + 1;
                }
                else results[i] = 1;
            }
            int sum = results[arr.size() - 1];
            for (int i = arr.size() - 2; i >= 0; i--){
                if (arr[i] > arr[i + 1] && results[i] <= results[i + 1]){
                    results[i] = results[i + 1] + 1;
                }
                sum += results[i];
            }
            return sum;
      }
```
