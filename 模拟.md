# 16.逆波兰表达式求值   
如果遇到数就入栈，如果遇到符号，先取出两个数然后运算，得到的结果再入栈。
```cpp
  int evalRPN(vector<string>& tokens) {
        stack<string> stk;
        unordered_set<string> hash = {"+", "-", "*", "/"};
        for (auto token : tokens){
            if (!hash.count(token)){
                stk.push(token);
            }
            else{
                int b = stoi(stk.top());
                stk.pop();
                int a = stoi(stk.top());
                stk.pop();
                switch(token[0]){      
                    case '+':
                        stk.push(to_string(a + b));
                        break;
                    case '-':
                        stk.push(to_string(a - b));
                        break;
                    case '*':
                        stk.push(to_string(a * b));
                        break;
                    case '/':
                        stk.push(to_string(a / b));
                }
            }
        }
        return stoi(stk.top());
    }
```
