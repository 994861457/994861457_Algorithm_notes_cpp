# 1.序列化二叉树   
采用的序列化思想，采用前序遍历，遇到空节点以#号保存，节点之间用“，”区分。如果遇到空节点之后都不保存了，也就是说这种策略会保存叶子节点下面的两个#，其他就没有#了。  
采用双指针来找到字符段并截取。使用引用的方式改变双指针的起始位置，即无论在哪里调用函数结束后都会使指针往下挪一段。也就是取字符串切片的顺序完全按照递归的顺序移动。  
最后还自己实现了字符串与数组的相互转换。也可以使用to_string和stoi。
```cpp
  class Solution {
  public:
      char* Serialize(TreeNode *root) {    
          string s;
          stack<TreeNode*> stk;
          stk.push(root);
          while (!stk.empty()){
              root = stk.top();
              stk.pop();
              if (!root) {
                  s = s + "#,";
              }
              else{
                  s = s + to_string(root -> val) + ",";
  //                 s = s + numToStr(root -> val) + ",";
                  stk.push(root -> right);
                  stk.push(root -> left);
              }
          }
          char* result = new char[s.size() + 1];
          strcpy(result, s.c_str());
          return result;
      }

      TreeNode* Deserialize(char *str) {
          string s(str);
          int index = 0;
          return createTree(s, index);
      }
  private:
      TreeNode* createTree(string s,int &index){
          int num = getNum(s, index);
          if (num == -1) return nullptr;
          TreeNode* newTree = new TreeNode(num);
          newTree -> left = createTree(s, index);
          newTree -> right = createTree(s, index);
          return newTree;
      }
      int getNum(string s, int &index){
          if (index >= s.size()) return -1;
          int fast = index;
          int slow = index;
          while (s[fast] != ',') fast++;
          string result = s.substr(slow, fast - slow);
          index = fast + 1;
          if (result == "#") return -1;
          int num = stoi(result);
  //         int num = strToNum(result);
          return num;
      }
```

# 2.合并k个已排序链表 NC51   
不断两两合并链表，直到合并到只剩下一个。注意在数量为奇数个的时候不要忘了最后一个。
```cpp  
  class Solution {
  public:
      ListNode *mergeKLists(vector<ListNode *> &lists) {
          if (lists.empty()) return {};
          while (lists.size() > 1){
              vector<ListNode*> temp;
              for (int i = 0; i < lists.size() / 2; i++){
                  temp.push_back(merge(lists[2 * i], lists[2 * i + 1]));
              }
              if (lists.size() % 2 == 1) temp.push_back(lists.back());
              lists = temp;
          }
          return lists[0];
      }
  private:
      ListNode* merge(ListNode* head1, ListNode* head2){
          ListNode* dummy = new ListNode(0);
          ListNode* curr = dummy;
          while (head1 && head2){
              if(head1 -> val <= head2 -> val){
                  curr -> next = head1;
                  head1 = head1 -> next;
              }
              else{
                  curr -> next = head2;
                  head2 = head2 -> next;
              }
              curr = curr -> next;
          }
          while (head1){
              curr -> next = head1;
              head1 = head1 -> next;
              curr = curr -> next;
          }
          while (head2){
              curr -> next = head2;
              head2 = head2 -> next;
              curr = curr -> next;
          }
          return dummy -> next;
      }
  };
```
