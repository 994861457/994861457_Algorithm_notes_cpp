# 1.  一组输入包括两个正整数a,b(1 <= a, b <= 10^9),输入数据有多组, 如果输入为0 0则结束输入。（数之间用空格分开，组之间用回车分开）。  
```cpp  
  #include<bits/stdc++.h> 
  using namespace std;

  int main(){
      int a, b;
      while (cin >> a >> b && (a || b)){ // cin读取以回车和空格为分隔（这俩字符不写入变量），cin读到文件末尾就会返回false
          cout << a + b << endl;
      }    
      return 0;
  }
```  
# 2.  输入数据多组，每组一行，开头第一个数字代表每组数字的长度， 开头第一个为0就结束。  
```cpp

   }

  int main(){
      int size;         //先读长度，然后再读后面的一组数
```

# 3.每组一行，没有其他的  
通过cin.get()判断‘\n’来分割一行  
```cpp
  #include<bits/stdc++.h>
  using namespace std;

  int main(){
      int temp;
      vector<int> input;
      while (cin >> temp){
          input.push_back(temp);
          if (cin.get() == '\n'){
              cout << accumulate(input.begin(), input.end(), 0) << endl;
              input.clear();
          }
      }
      return 0;
  }
```
# 4.字符串的输入，每行一组，每个字符串之间用空格分开。  
与数组类似的解法
```cpp  
  #include<bits/stdc++.h>  
  using namespace std;

  void output(const vector<string> &input){
      for (int i = 0; i < input.size(); i++){
          cout << input[i];
          if (i != input.size() - 1){
          cout << " ";
      }    
      }
      cout << endl;
  }

  int main(){
      string temp;
      vector<string> input;
      while(cin >> temp){
          input.push_back(temp);
          if (cin.get() == '\n'){
              sort(input.begin(), input.end());
              output(input);

              input.clear();
          }
      }
      return 0;
  }
```
# 5.字符串的输入，每行一组，每个字符串之间用逗号分开。   
有符号分隔可以使用getline更方便
```cpp
  #include<bits/stdc++.h>
  using namespace std;

  int main(){
      string s;
      string temp;
      vector<string> input;
      while(getline(cin, s)){    //大循环一行一行的读取，保存到字符串变量中
          stringstream ss;   //类似于cin与cout的流，只不过这个是一个变量。
          ss << s;            //将读到的一行放入流中。
          while(getline(ss, temp, ',')){        //放入流中是为了能继续使用getline。
              input.push_back(temp);
          }
          sort(input.begin(), input.end());
          for (int i = 0; i < input.size() - 1; i++){
                  cout << input[i] << ',';
              }
           cout << input.back() << endl; 
           input.clear();   
      }
      return 0;
  }
```  
# 6.IP地址转化 NC337  
```cpp  
  string IPtoNum(string ip) {
          stringstream s;   //(流)创建这个变量是为了能够使用getline去方便的提取点分的字符.
          s << ip;            //
          string temp;
          vector<string> input;
          while (getline(s, temp, '.')){
              input.push_back(temp);
          }
          long result = 0;
          for (int i = 0; i < 4; i++){
              int r = stoi(input[i]);
              result = result * 256 + r;
          }
          string rs = to_string(result);
          return rs;
      }
```
